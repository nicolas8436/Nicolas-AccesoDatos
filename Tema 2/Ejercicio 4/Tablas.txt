CREATE TABLE FACTURAS (
    NUMERO          NUMBER PRIMARY KEY,
    NOMBRE_CLIENTE  VARCHAR2(100) NOT NULL,
    FECHA_EMISION   DATE DEFAULT SYSDATE,
    DIRECCION       VARCHAR2(100),
    CIF_CLIENTE     VARCHAR2(20)
);

CREATE TABLE APUNTES (
    ID              NUMBER PRIMARY KEY,
    NUMERO_FACTURA  NUMBER NOT NULL,
    PRODUCTO        VARCHAR2(100) NOT NULL,
    CANTIDAD        NUMBER(10,2) NOT NULL,
    PRECIO_UNIDAD   NUMBER(10,2) NOT NULL,
    DESCRIPCION     VARCHAR2(200),
    IVA             NUMBER(5,2) DEFAULT 21,
    DESCUENTO       NUMBER(5,2) DEFAULT 0,
    CONSTRAINT FK_APUNTES_FACTURA FOREIGN KEY (NUMERO_FACTURA)
        REFERENCES FACTURAS (NUMERO)
        ON DELETE CASCADE
);
